## Schema

```typescript
interface TimelineItem {
  time?: string;                         // Item timestamp/date
  title?: string;                        // Item title
  description?: string;                  // Item description
  content?: SchemaNode | SchemaNode[];   // Additional content
  icon?: string;                         // Icon or emoji
  variant?: 'default' | 'success' | 'warning' | 'error';  // Marker variant
  className?: string;                    // Item CSS classes
}

interface TimelineSchema {
  type: 'timeline';
  
  // Layout
  variant?: 'vertical' | 'horizontal' | 'gantt';  // Timeline layout (default: 'vertical')
  
  // Items
  items: TimelineItem[];                 // Timeline items
  
  // Date formatting
  dateFormat?: 'short' | 'long' | string; // Date display format (default: 'short')
  
  // Gantt-specific (when variant='gantt')
  minDate?: string;                      // Gantt chart start date
  maxDate?: string;                      // Gantt chart end date
  timeScale?: 'day' | 'week' | 'month';  // Gantt time scale
  
  // Events
  onItemClick?: (item: any) => void;    // Item click handler
  
  // Styling
  className?: string;                    // Tailwind CSS classes
  
  // Base properties
  id?: string;
  visible?: boolean | string;
  testId?: string;
}
```
