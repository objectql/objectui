## Schema

```typescript
interface CalendarSchema {
  type: 'calendar';
  mode?: 'single' | 'multiple' | 'range';
  selected?: Date | Date[] | { from: Date; to: Date };
  
  // Events
  onSelect?: string | ActionConfig;
  
  // States
  disabled?: boolean | Date[];    // Disabled dates
  
  // Styling
  className?: string;
}
```
