The Resizable component allows users to resize panels by dragging handles between them.

## Basic Usage

<ComponentDemo
  schema={{
    type: 'resizable',
    direction: 'horizontal',
    panels: [
      {
        defaultSize: 50,
        content: {
          type: 'card',
          content: 'Panel 1'
        }
      },
      {
        defaultSize: 50,
        content: {
          type: 'card',
          content: 'Panel 2'
        }
      }
    ]
  }}
  title="Horizontal Panels"
/>

## Schema

```typescript
interface ResizablePanel {
  defaultSize?: number;           // Default size (percentage)
  minSize?: number;               // Minimum size
  maxSize?: number;               // Maximum size
  content: ComponentSchema;       // Panel content
}

interface ResizableSchema {
  type: 'resizable';
  direction: 'horizontal' | 'vertical';
  panels: ResizablePanel[];       // Panels to display
  className?: string;
}
```
